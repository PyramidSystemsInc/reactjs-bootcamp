Some things to try:
  • Ternary Operator 
    var x = 1 === 2 ? 5 : 4;
  • Active Element 
    var x = document.activeElement;
  • Get Selection 
    var x = getSelection();
  • Some loop 
    [1, 2, 3, 4, 5].some(function(number) {
      console.log(number);
      return number === 3;
    }.bind(this));
  • Map Function 
    var x = [1, 2, 3, 4, 5].map(function(number) {
      return number;
    }.bind(this));
  • ForEach Loop 
    [1, 2, 3, 4, 5].forEach(function(number) {
      console.log(number);
    }.bind(this));
  • Arrow Functions 
    // Note this is the same as the code in the "ForEach Loop" section above
    [1, 2, 3, 4, 5].forEach((number) => console.log(number));
  • Creating Promises 
    var x = new Promise(function(resolve, reject) {
      var delay = 20000; // 20 seconds
      setTimeout(function() {
        resolve('Delay is up. I just resolved');
      }.bind(this), delay);
    });
  • Using Resolved Promises 
    x.then(function(value) {
      console.log(value);
    }.bind(this));
  • "Or" Defaults 
    var y;
    var x = y || 2;
  • HTML 5 DataSet 
    // Look in index.html to understand
    var listItem = document.getElementById('dataset-example');
    var x = listItem.dataset.something;
  • QuerySelectorAll Function 
    // Look in index.html to understand
    var container = document.getElementById('container');
    var x = container.querySelectorAll('li[data-title]');
